### Combien de poll() dans le code :
# UN SEUL poll() dans tout le projet.
# Il se trouve dans Server.cpp, dans la fonction Server::run() :

#   int ret = poll(&_pollfds[0], _pollfds.size(), -1);

### poll() avant chaque accept/recv/send :
# - accept() : appele dans _acceptClient(), qui n'est executee QUE quand
#   poll() a signale POLLIN sur la socket d'ecoute (_listenFd)
#
# - recv() : appele dans _receiveData(), qui n'est executee QUE quand
#   poll() a signale POLLIN sur le fd du client
#
# - send() : appele dans Client::flushSendBuffer(), qui est appele par
#   Server::_sendData(), qui n'est executee QUE quand poll() a signale
#   POLLOUT sur le fd du client
#
# AUCUN send() direct n'est fait ailleurs dans le code.
# Tous les messages passent par queueMessage() -> buffer -> POLLOUT -> send()

### Pas d'utilisation de errno :
# Apres accept(), recv() et send(), on ne verifie JAMAIS errno.
# - Si recv() retourne <= 0 : on marque le client pour deconnexion
# - Si send() retourne < 0 : on marque le client pour deconnexion
# - Si accept() retourne < 0 : on return simplement
# Aucun "if (errno == EAGAIN)" dans le code.
# Pour le prouver :
#   grep -rn "errno" src/
# -> Aucun resultat

### fcntl() :
# Deux appels a fcntl() dans tout le code, les deux identiques :

## Nos reponses :

### Commandes partielles :
# Le mecanisme est dans Client::extractLine()
# Fichier : src/Client.cpp, ligne ~78
#
# Fonctionnement :
# - recv() ajoute les donnees brutes au buffer de reception (_buffer)
# - extractLine() cherche \r\n (ou \n) dans le buffer
# - Si pas de \r\n : retourne "" -> on ne traite rien, on attend la suite
# - Si \r\n trouve : retourne la ligne complete et l'efface du buffer
#
# Donc si un client envoie "PRIV" puis "MSG #chan :hi\r\n" en 2 paquets :
#   1er recv : _buffer = "PRIV" -> extractLine() = "" (pas de \r\n, on attend)
#   2e recv  : _buffer = "PRIVMSG #chan :hi\r\n" -> extractLine() = "PRIVMSG #chan :hi"
#
# Pendant ce temps, les autres connexions fonctionnent normalement car
# poll() continue de surveiller TOUS les fd.
#
# Pour tester avec nc :
#   printf "PASS pas" | nc 127.0.0.1 6667 &
#   # (commande partielle envoyee, pas de \r\n)
#   # Ouvrir un autre nc et verifier qu'il fonctionne normalement
#   nc 127.0.0.1 6667
#   PASS pass123
#   NICK test
#   USER test 0 * :Test
#   # -> Ca marche, le serveur repond normalement

### Kill inattendu d'un client :
# Quand un client est kill (kill -9, fermeture du terminal, Ctrl+C sur nc) :
# - Le fd est ferme cote client
# - poll() detecte POLLHUP ou POLLERR sur ce fd
# - Fichier : src/Server.cpp, dans la boucle run(), ligne ~161
#     if (rev & (POLLERR | POLLHUP | POLLNVAL)) {
#         _toRemove.insert(fd);
#     }
# - Le client est proprement deconnecte dans _removeClient() :
#   * Retire de tous les channels
#   * Broadcast QUIT aux autres membres
#   * close(fd), delete client, retire du tableau poll
# - Le serveur continue de fonctionner pour les autres clients
#
# Pour tester :
#   Terminal 1 : ./ircserv 6667 pass123
#   Terminal 2 : nc 127.0.0.1 6667 (se connecter, JOIN #test)
#   Terminal 3 : nc 127.0.0.1 6667 (se connecter, JOIN #test)
#   -> Kill le terminal 2 (Ctrl+C ou fermer le terminal)
#   -> Terminal 3 recoit ":nick!user@host QUIT :Leaving"
#   -> Le serveur affiche "Client disconnected (fd X)"
#   -> Terminal 3 fonctionne toujours normalement
#   -> Un nouveau nc peut se connecter sans probleme

### Kill nc avec commande partielle :
# Meme mecanisme que ci-dessus. Si nc est kill avec un buffer partiel :
# - Le fd est ferme -> poll() detecte POLLHUP
# - _removeClient() est appele
# - Le buffer partiel (_buffer) est simplement libere avec le delete du Client
# - Aucun traitement de la commande incomplete
# - Le serveur n'est pas dans un etat bizarre
#
# Fichier : src/Server.cpp, _removeClient(), ligne ~370
# Le delete du Client libere automatiquement le _buffer (std::string)
#
# Pour tester :
#   Terminal 2 : nc 127.0.0.1 6667
#   PASS pass123
#   NICK test
#   USER test 0 * :Test
#   (taper "PRIVMSG #te" sans appuyer Entree, puis Ctrl+C)
#   -> Le serveur detecte la deconnexion, cleanup propre
#   -> Les autres clients fonctionnent toujours

### Client stoppe (^Z) + flood :
# C'est le point le plus critique. Notre systeme de buffer d'envoi le gere.
#
# Quand un client est stoppe (^Z = SIGTSTP) :
# - Son processus est suspendu, il ne lit plus sur sa socket
# - Son buffer TCP kernel se remplit progressivement
# - MAIS notre serveur ne fait JAMAIS de send() bloquant
#
# Mecanisme (le plus important du projet) :
# 1. Les messages sont stockes dans _sendBuffer du client (Client.cpp)
#    Fichier : src/Client.cpp, queueMessage(), ligne ~93
#
# 2. poll() surveille POLLOUT sur ce fd
#    Fichier : src/Server.cpp, mise a jour des events, ligne ~192
#
# 3. Quand POLLOUT est signale, on tente d'envoyer via flushSendBuffer()
#    Fichier : src/Client.cpp, flushSendBuffer(), ligne ~112
#    - send() en non-bloquant envoie ce qu'il peut
#    - Si le buffer kernel est plein : send() retourne le nombre d'octets
#      effectivement envoyes (possiblement 0 ou partiel)
#    - On efface seulement ce qui a ete envoye, le reste attend
#
# 4. Si le buffer kernel est completement plein :
#    - POLLOUT n'est plus signale par poll() (le fd n'est pas pret en ecriture)
#    - Les messages s'accumulent dans _sendBuffer
#    - Le serveur continue de tourner normalement pour les autres clients
#
# 5. Quand le client reprend (fg) :
#    - Il recommence a lire sa socket -> le buffer kernel se vide
#    - poll() signale a nouveau POLLOUT
#    - flushSendBuffer() envoie les messages accumules
#    - Le client recoit tous les messages qu'il a manques
#
# Le serveur ne hang JAMAIS car :
# - send() est non-bloquant (fd en O_NONBLOCK)
# - On ne send() que quand poll() dit POLLOUT
# - Si POLLOUT n'est pas signale, on ne tente meme pas d'envoyer
#
# Pour tester :
#   Terminal 1 : ./ircserv 6667 pass123
#   Terminal 2 : nc 127.0.0.1 6667
#     PASS pass123 / NICK alice / USER alice 0 * :Alice / JOIN #flood
#   Terminal 3 : nc 127.0.0.1 6667
#     PASS pass123 / NICK bob / USER bob 0 * :Bob / JOIN #flood
#   -> Stopper terminal 2 : Ctrl+Z
#   -> Dans terminal 3, envoyer plein de messages :
#     PRIVMSG #flood :msg1
#     PRIVMSG #flood :msg2
#     PRIVMSG #flood :msg3
#     ... (repeter beaucoup)
#   -> Le serveur ne bloque pas, terminal 3 fonctionne toujours
#   -> Reprendre terminal 2 : fg
#   -> alice recoit tous les messages accumules d'un coup

### Memory leaks :
# Pour verifier les leaks pendant le test ^Z + flood :
#   valgrind --leak-check=full ./ircserv 6667 pass123
# Ou sur macOS :
#   leaks --atExit -- ./ircserv 6667 pass123
#
# Pas de leak car :
# - Les messages sont dans des std::string (_sendBuffer) -> liberes automatiquement
# - Pas de new/malloc pour les messages, juste de la concatenation de string
# - Le _sendBuffer est libere quand le Client est delete (deconnexion)
